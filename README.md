# Товары

```ts
type Product = {
  id: string;
  name: string;
  description: string;
  img: string;
  type:
    | "tables"
    | "other"
    | "kitchen"
    | "print"
    | "material"
    | "table"
    | "closet";
  price: number;
  params: {
    width: number;
    height: number;
    material: number;
    weight: number;
  };
};
```
